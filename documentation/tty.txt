TTY - Teletype or Terminal

TTY Colors:
        TTY_BLACK = 0,
	TTY_BLUE = 1,
	TTY_GREEN = 2,
	TTY_CYAN = 3,
	TTY_RED = 4,
	TTY_MAGENTA = 5,
	TTY_BROWN = 6,
	TTY_LIGHT_GREY = 7,
	TTY_GREY = 8,
	TTY_LIGHT_BLUE = 9,
	TTY_LIGHT_GREEN = 10,
	TTY_LIGHT_CYAN = 11,
	TTY_LIGHT_RED = 12,
	TTY_LIGHT_MAGENTA = 13,
	TTY_LIGHT_BROWN = 14,
	TTY_WHITE = 15

TTY Functions:

void tty_init(enum tty_color fg, enum tty_color bg);
	Purpose: Initializes TTY
	fg = Foreground color
	bg = Background color

void tty_putc(char c, enum tty_color fg, enum tty_color bg);
	Purpose: Puts character onto TTY buffer
	c = Character
	fg = Foreground color
	bg = Background color

void tty_puts(char *c, enum tty_color fg, enum tty_color bg);
	Purpose: Puts string of characters onto TTY buffer
	c = String
	fg = Foreground color
	bg = Background color

void tty_printf(char *str, ...);
	Purpose: Prints a formatted string
	str = String
	... = Additional arguments
